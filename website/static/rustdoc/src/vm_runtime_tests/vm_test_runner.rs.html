use crate::{
    fake_data_cache::{FakeDataCache, GENESIS_WRITE_SET, TESTNET_GENESIS_WRITE_SET},
    Account, AccountData,
};
use storage_client::StateView;
use types::{
    transaction::{SignedTransaction, TransactionOutput},
    vm_error::VMStatus,
    write_set::WriteSet,
};
use vm_runtime::{
    loaded_data::{struct_def::StructDef, types::Type},
    value::Value,
    LibraVM, VMExecutor, VMVerifier,
};

/// Provides an environment to run a VM instance via execute_block().
/// It mocks an Executor
#[derive(Debug, Default)]
pub struct FakeExecutor {
    data_store: FakeDataCache,
}

impl FakeExecutor {
    // Create a number of Accounts all with the same balance and sequence number.
    // Publish the Accounts to the Executor
    pub fn create_accounts(&mut self, size: usize, balance: u64, seq_num: u64) -> Vec<Account> {
        let mut accounts: Vec<Account> = Vec::with_capacity(size);
        for _i in 0..size {
            let account_data = AccountData::new(balance, seq_num);
            self.add_account_data(&account_data);
            accounts.push(account_data.into_account());
        }
        accounts
    }

    /// Create a FakeExecutor from a genesis WriteSet
    pub fn from_genesis(write_set: &WriteSet) -> Self {
        let mut executor = FakeExecutor::default();
        executor.apply_write_set(write_set);
        executor
    }

    /// Create a FakeExecutor from the genesis file GENESIS_FILE_LOCATION
    pub fn from_genesis_file() -> Self {
        Self::from_genesis(&GENESIS_WRITE_SET)
    }

    /// Create a FakeExecutor from the genesis file GENESIS_FILE_LOCATION
    pub fn from_testnet_genesis_file() -> Self {
        Self::from_genesis(&TESTNET_GENESIS_WRITE_SET)
    }

    /// Apply a WriteSet to the data store
    pub fn apply_write_set(&mut self, write_set: &WriteSet) {
        self.data_store.add_write_set(write_set);
    }

    /// Helper to add account to the storage for this FakeExecutor
    pub fn add_account_data(&mut self, account_data: &AccountData) {
        self.data_store.add_account_data(account_data)
    }

    /// Read the data for an account
    pub fn read_account_resource(&self, account: &Account) -> Option<Value> {
        let ap = account.make_access_path();
        let data_blob = StateView::get(&self.data_store, &ap)
            .expect("account must exists in data store")
            .expect("data must exists in data store");
        let account_type = Self::get_account_struct_def();
        Account::read_account_resource(&data_blob, account_type)
    }

    /// Execute a block of transactions.
    /// A test would call it and check the expected result of each TransactionOutput.
    /// For each WriteSet it should call apply_write_set() to update storage.
    pub fn execute_block(&self, txn_block: Vec<SignedTransaction>) -> Vec<TransactionOutput> {
        LibraVM::execute_block(txn_block, &self.data_store)
    }

    /// Verify a transactions.
    pub fn verify_transaction(&self, txn: SignedTransaction) -> Option<VMStatus> {
        let vm = LibraVM::new();
        vm.validate_transaction(txn, &self.data_store)
    }

    /// TODO: This is a hack and likely to break soon. THe Account type is replicated here with no
    /// checks that is the right now. Fix it!
    fn get_account_struct_def() -> StructDef {
        // STRUCT DEF StructDef(StructDefInner { field_definitions: [ByteArray,
        // Struct(StructDef(StructDefInner { field_definitions: [Integer] })), Integer, Integer,
        // Integer] }) let currency = StructDef(StructDefInner { field_definitions:
        // [Type::Integer] })
        let int_type = Type::Integer;
        let byte_array_type = Type::ByteArray;
        let currency = Type::Struct(StructDef::new(vec![int_type.clone()]));
        StructDef::new(vec![
            byte_array_type,
            currency,
            int_type.clone(),
            int_type.clone(),
            int_type.clone(),
        ])
    }
}
