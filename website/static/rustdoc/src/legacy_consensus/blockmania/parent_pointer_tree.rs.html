use std::{
    fmt,
    hash::{Hash, Hasher},
    sync::Arc,
};

#[cfg(test)]
#[path = "unit_tests/parent_pointer_tree_tests.rs"]
mod parent_pointer_tree_tests;

/// An immutable parent pointer node. May be empty or contain a value; may have a pointer to a
/// parent or not.
///
/// A parent pointer tree is a (possibly empty) node with a (possibly null) pointer to a parent
/// node. Any given node has a unique path back to the root node. Rather than mutably updating the
/// stack, one creates and obtains access to immutable nodes (when a node becomes unreachable its
/// memory is automatically reclaimed). A new child node pointing to a parent can be created via the
/// `child` function (analogous to the normal `push`) and a parent can be retrieved via the `parent`
/// function (analogous to the normal `pop`).
#[derive(Clone, Default)]
pub struct ParentPointerTree<T> {
    node: Option<Arc<Node<T>>>,
}

#[derive(Clone)]
struct Node<T> {
    val: T,
    parent: Option<Arc<Node<T>>>,
}

impl<T> ParentPointerTree<T> {
    /// Return an empty parent pointer tree node.
    pub fn new() -> ParentPointerTree<T> {
        ParentPointerTree { node: None }
    }

    /// Is this parent pointer tree node empty?
    pub fn is_empty(&self) -> bool {
        self.node.is_none()
    }

    /// How many items are there in this parent pointer tree?
    pub fn len(&self) -> usize {
        self.vals().count()
    }

    /// Create a new parent pointer tree node containing value `val` and pointing to parent `self`.
    pub fn child(&self, val: T) -> ParentPointerTree<T> {
        ParentPointerTree {
            node: Some(Arc::new(Node {
                val,
                parent: self.node.clone(),
            })),
        }
    }

    /// Return this parent pointer tree node's parent node or `None` if this parent pointer tree is
    /// empty.
    pub fn parent(&self) -> Option<ParentPointerTree<T>> {
        self.node.as_ref().map(|n| ParentPointerTree {
            node: n.parent.clone(),
        })
    }

    /// Return a reference to this parent pointer tree node's value or `None` if this parent pointer
    /// tree is empty.
    pub fn val(&self) -> Option<&T> {
        self.node.as_ref().map(|n| &n.val)
    }

    /// Return an iterator over this parent pointer tree's values. Note that the iterator produces
    /// values starting from this node and then walking up towards the root.
    pub fn vals(&self) -> ParentPointerTreeValsIter<'_, T> {
        ParentPointerTreeValsIter {
            next: self.node.as_ref(),
        }
    }

    /// Try to consume this ParentPointerTree node and return its data. If the parent pointer tree
    /// node has no children, this succeeds; if the parent pointer tree node has children, it
    /// fails, and returns the original parent pointer tree node.
    pub fn try_unwrap(self) -> Result<T, ParentPointerTree<T>> {
        match self.node {
            None => Err(ParentPointerTree { node: None }),
            Some(x) => match Arc::try_unwrap(x) {
                Ok(n) => Ok(n.val),
                Err(rc) => Err(ParentPointerTree { node: Some(rc) }),
            },
        }
    }
}

/// An iterator over a `ParentPointerTree` stack's values. Note that the iterator produces values
/// starting from this node and then walking up towards the root.
pub struct ParentPointerTreeValsIter<'a, T> {
    next: Option<&'a Arc<Node<T>>>,
}

impl<'a, T> Iterator for ParentPointerTreeValsIter<'a, T> {
    type Item = &'a T;

    fn next(&mut self) -> Option<Self::Item> {
        self.next.take().map(|n| {
            self.next = n.parent.as_ref();
            &n.val
        })
    }
}

impl<T: PartialEq> PartialEq for ParentPointerTree<T> {
    fn eq(&self, other: &ParentPointerTree<T>) -> bool {
        // This is, in a sense, a manually expanded self.vals().zip(other.vals()) -- doing so
        // allows us to potentially shortcut the checking of every element using Arc::ptr_eq.
        let mut si = self.node.as_ref();
        let mut oi = other.node.as_ref();
        while si.is_some() && oi.is_some() {
            let sn = si.unwrap();
            let on = oi.unwrap();
            // If we're lucky, the two Arc's are pointer equal, proving that the two cactuses are
            // equal even without ascending the parent hierarchy.
            if Arc::ptr_eq(sn, on) {
                return true;
            }
            if sn.val != on.val {
                return false;
            }
            if let Some(n) = si.take() {
                si = n.parent.as_ref();
            }
            if let Some(n) = oi.take() {
                oi = n.parent.as_ref();
            }
        }
        // If one of the iterators finished before the other, the two cactuses were of different
        // length and thus unequal by definition; otherwise they were equal.
        !(si.is_some() || oi.is_some())
    }
}

impl<T: Eq> Eq for ParentPointerTree<T> {}

impl<T: Hash> Hash for ParentPointerTree<T> {
    fn hash<H: Hasher>(&self, state: &mut H) {
        for v in self.vals() {
            v.hash(state);
        }
    }
}

impl<T: fmt::Debug> fmt::Debug for ParentPointerTree<T> {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "ParentPointerTree[")?;
        for (i, x) in self.vals().enumerate() {
            if i > 0 {
                write!(f, ", ")?;
            }
            write!(f, "{:?}", x)?;
        }
        write!(f, "]")
    }
}
