//! Implement modified Patricia Merkle tree as specified by
//! https://github.com/ethereum/wiki/wiki/Patricia-Tree.

use crate::{
    nibbler::Nibbler,
    node_type::{BranchType, ExtensionType, LeafType, NodeType},
    schema_batch_wrapper::{get_node_from_db, get_root_hash_from_db, SchemaBatchWrapper},
};
use crypto::{
    self,
    hash::{CryptoHash, TestOnlyHasher, DEFAULT_DIGEST},
    HashValue,
};
use failure::prelude::*;
use libra_store::{
    db::{SchemaBatchWithIndex, DB},
    schema::Schema,
};
use std::{convert::TryFrom, sync::Arc};
use types::{proof::SparseMerkleProofDeprecated, transaction::Version};

const SHARD_ROOT_NIBBLE_HEIGHT: u8 = 63;
/// The limit of maximum number of invalid digests for a conditional validation
/// See ModifiedPatriciaMerkleTrie::validate() for details.
const MAX_INVALID_NODE: usize = 100;

#[derive(Debug, PartialEq, Eq)]
pub enum Validity {
    Valid,
    Invalid,
    ConditionalValid(Vec<HashValue>),
}

/// Modified Patricia Merkle Trie implementation. `S` is the schema of the underlying DB that
/// stores the modified patricia merkle trie.
#[derive(Debug)]
pub struct ModifiedPatriciaMerkleTrie<S> {
    db: Arc<DB<S>>,
}

/// The proof is a series of digests of non-default siblings with a
/// bitmap indicating their height in the trie.
/// The verification process will be bottom-up, similar to trie node
/// digest generation process
pub fn verify(
    proof: &SparseMerkleProofDeprecated,
    mut access_path_nibbler: Nibbler,
    value: &[u8],
    root_digest: &HashValue,
) -> Result<bool> {
    let mut digest = NodeType::leaf_digest(TestOnlyHasher::default(), value);
    let mut non_default_siblings = proof.non_default_siblings.iter();
    let mut bitmap_nibbler = Nibbler::new_with_num_nibbles(
        proof.bitmap.to_vec(),
        access_path_nibbler.num_all_remaining(),
    );
    while let Some(nibble) = access_path_nibbler.next_back() {
        let bitmap = match bitmap_nibbler.next_back() {
            Some(bitmap) => bitmap,
            None => return Ok(false),
        };
        for i in 0..4 {
            let sibling = if bitmap >> i & 1 != 0 {
                // There is a non-default sibling digest on this height
                match non_default_siblings.next_back() {
                    Some(sibling) => sibling,
                    None => return Ok(false),
                }
            } else {
                &DEFAULT_DIGEST
            };

            digest = if nibble >> i & 0x01 == 0 {
                NodeType::interior_digest(TestOnlyHasher::default(), &digest, sibling)
            } else {
                NodeType::interior_digest(TestOnlyHasher::default(), sibling, &digest)
            }
        }
    }

    Ok(digest == *root_digest)
}

impl<S> ModifiedPatriciaMerkleTrie<S>
where
    S: PatriciaMerkleSchema,
{
    /// Constructs a new `ModifiedPatriciaMerkleTrie` instance.
    pub fn new(db: Arc<DB<S>>) -> Self {
        ModifiedPatriciaMerkleTrie { db }
    }

    /// Puts a list of key/value pairs into the P-M tree. All of them are supposed to come from a
    /// single transaction. Returns a `SchemaBatchWithIndex` and the new root hash.  The caller is
    /// supposed to take the return value and write additional info about root node and latest
    /// version with a version number into the write batch and send it to DB.
    pub fn put<T>(&self, kvs: Vec<(Nibbler, T)>) -> Result<(SchemaBatchWithIndex<S>, HashValue)>
    where
        T: Into<Vec<u8>>,
        Vec<u8>: From<T>,
    {
        let mut batch = SchemaBatchWrapper::new(Arc::clone(&self.db));

        // Get the most recent root hash in case `kvs` is an empty list.
        let mut root_hash = batch.get_root_hash()?;

        // Insert all key/value pairs and update root hash.
        for (nibbler, value) in kvs {
            let new_root_hash = self.put_one(nibbler, value, &mut batch)?;
            batch.set_root_hash(new_root_hash);
            root_hash = new_root_hash;
        }
        Ok((batch.into_schema_batch(), root_hash))
    }

    /// Used to insert a key/value pair into the P-M tree.
    fn put_one<T>(
        &self,
        nibbler: Nibbler,
        value: T,
        batch: &mut SchemaBatchWrapper<S>,
    ) -> Result<HashValue>
    where
        T: Into<Vec<u8>>,
        Vec<u8>: From<T>,
    {
        ensure!(
            nibbler.num_all_remaining() as u8 == SHARD_ROOT_NIBBLE_HEIGHT,
            "Current implementation does not support variable nibbler remaining \
             length. nibbler key: {:?}, nibbler remaining length: {},  \
             shard_root_nibble_height: {}",
            nibbler.encoded_path(),
            nibbler.num_all_remaining(),
            SHARD_ROOT_NIBBLE_HEIGHT
        );

        // Get root node. If this is the first operation, it would get root node from db. Otherwise
        // it most likely would come from the schema batch.
        let mut root = batch.get_root_node()?;

        // Do insertion on root node.
        let new_root_hash = self.insert(&mut root, nibbler, value.into(), batch)?;
        Ok(new_root_hash)
    }

    /// Helper function for recursive insertion into the subtree that starts from the current node
    /// `cur_node`. Returns hash of the created node.
    ///
    /// It is safe to use recursion here because the max depth is limited by the key length which
    /// for this tree will be the hash of an account.  We do the input size validation in the
    /// `put_one` function which calls `insert`.
    fn insert(
        &self,
        cur_node: &mut NodeType,
        nibbler: Nibbler,
        insert_value: Vec<u8>,
        batch: &mut SchemaBatchWrapper<S>,
    ) -> Result<HashValue> {
        match cur_node {
            NodeType::Null(_) => {
                // If we are on a null node, we need to create a new leaf node with the rest
                // of the key as our encoded_path and replace the null node with leaf node.
                self.insert_leaf_node(cur_node, nibbler, insert_value, batch)
            }
            NodeType::Leaf(_) => {
                // We are on a leaf node but trying to insert another node, so we may need to
                // diverge.
                self.diverge_from_leaf_node_and_insert(cur_node, nibbler, insert_value, batch)
            }
            NodeType::Branch(_) => {
                self.follow_branch_and_insert(cur_node, nibbler, insert_value, batch)
            }
            NodeType::Extension(_) => {
                self.follow_extension_and_insert(cur_node, nibbler, insert_value, batch)
            }
        }
    }

    fn follow_extension_and_insert(
        &self,
        cur_node: &mut NodeType,
        mut nibbler: Nibbler,
        insert_value: Vec<u8>,
        batch: &mut SchemaBatchWrapper<S>,
    ) -> Result<HashValue> {
        debug_assert!(
            cur_node.is_extension(),
            "Expecting extension node. Current node tag: {}.",
            cur_node.tag()
        );
        debug_assert!(
            !cur_node.extension()?.encoded_path.is_empty(),
            "Extension node should not have empty encoded_path."
        );

        let mut extension_nibbler = Nibbler::new_with_is_odd(
            cur_node.extension()?.encoded_path.clone(),
            cur_node.extension()?.odd_encoded_path,
        );

        // First see how much of the extension matches our current key.
        let mut matching_nibbles = nibbler.get_matching_nibbles(&mut extension_nibbler);

        if !extension_nibbler.is_finished() {
            // Only some or no nibbles matched so diverge.
            return self.diverge_path(
                cur_node,
                nibbler,
                insert_value,
                &mut matching_nibbles,
                batch,
            );
        }

        debug_assert!(
            !nibbler.is_finished(),
            "We should never end the search on an extension node when key length is fixed."
        );

        // This means we can follow the entire extension, so grab the next node and recurse with
        // it.
        let mut next_node = batch.get_node(&cur_node.extension()?.child)?;
        debug_assert!(
            next_node.is_branch(),
            "The child of an extension node should always be a branch node. Child node tag: {}.",
            next_node.tag()
        );
        let inserted_child_hash = self.insert(&mut next_node, nibbler, insert_value, batch)?;

        // Update the extension node.
        let old_node_hash = cur_node.hash();
        cur_node.extension_mut()?.child = inserted_child_hash;
        batch.replace_node(&old_node_hash, cur_node.clone())?;
        Ok(cur_node.hash())
    }

    fn insert_leaf_node(
        &self,
        cur_node: &mut NodeType,
        mut nibbler: Nibbler,
        insert_value: Vec<u8>,
        batch: &mut SchemaBatchWrapper<S>,
    ) -> Result<HashValue> {
        debug_assert!(
            cur_node.is_null(),
            "Expecting null node. Current node tag: {}.",
            cur_node.tag()
        );

        // Get all remaining nibbles and use them as encoded path.
        let nibbles = nibbler.get_all_remaining();

        // Now create the new leaf node.
        let new_leaf = NodeType::Leaf(LeafType {
            encoded_path: nibbles.encoded_path().to_vec(),
            odd_encoded_path: nibbles.is_odd(),
            value: insert_value,
        });

        // Replace the null node with the new leaf node and return the hash of leaf.
        batch.replace_node(&cur_node.hash(), new_leaf.clone())?;
        *cur_node = new_leaf;
        Ok(cur_node.hash())
    }

    fn follow_branch_and_insert(
        &self,
        cur_node: &mut NodeType,
        mut nibbler: Nibbler,
        insert_value: Vec<u8>,
        batch: &mut SchemaBatchWrapper<S>,
    ) -> Result<HashValue> {
        debug_assert!(
            cur_node.is_branch(),
            "Expecting branch node. Current node tag: {}.",
            cur_node.tag(),
        );

        // Follow the branch. Check if there is a node with this index.
        let index: u8 = match nibbler.next() {
            Some(index) => index,
            _ => bail!("Nibbler ran out of nibbles"),
        };
        let mut next_node = match cur_node.branch()?.children.get(&index) {
            // Grab next node.
            Some(child) => batch.get_node(child)?,
            None => NodeType::default(),
        };

        // Recurse w/ this new node.
        let inserted_child_hash = self.insert(&mut next_node, nibbler, insert_value, batch)?;

        // Modify the current branch node to generate the new branch node.
        let old_node_hash = cur_node.hash();
        cur_node.set_child(index, inserted_child_hash)?;
        batch.replace_node(&old_node_hash, cur_node.clone())?;
        Ok(cur_node.hash())
    }

    fn diverge_from_leaf_node_and_insert(
        &self,
        cur_node: &mut NodeType,
        mut nibbler: Nibbler,
        insert_value: Vec<u8>,
        batch: &mut SchemaBatchWrapper<S>,
    ) -> Result<HashValue> {
        debug_assert!(
            cur_node.is_leaf(),
            "Expecting leaf node. Current node tag: {}.",
            cur_node.tag(),
        );

        // First determine the common portions between this leaf node and the incoming key.
        let mut leaf_nibbler = Nibbler::new_with_is_odd(
            cur_node.leaf()?.encoded_path.clone(),
            cur_node.leaf()?.odd_encoded_path,
        );
        let mut matching_nibbles = nibbler.get_matching_nibbles(&mut leaf_nibbler);
        assert_eq!(
            nibbler.is_finished(),
            leaf_nibbler.is_finished(),
            "Current implementation does not support variable key length."
        );

        if !nibbler.is_finished() && !leaf_nibbler.is_finished() {
            // Both did not finish. We need to diverge.
            return self.diverge_path(
                cur_node,
                nibbler,
                insert_value,
                &mut matching_nibbles,
                batch,
            );
        }

        // Both finished. That means the incoming key already exists in the trie and we just need
        // to update its value.
        let old_node_hash = cur_node.hash();
        cur_node.leaf_mut()?.value = insert_value;
        batch.replace_node(&old_node_hash, cur_node.clone())?;
        Ok(cur_node.hash())
    }

    fn diverge_path(
        &self,
        cur_node: &mut NodeType,
        mut nibbler: Nibbler,
        insert_value: Vec<u8>,
        matching_nibbles: &mut Nibbler,
        batch: &mut SchemaBatchWrapper<S>,
    ) -> Result<HashValue> {
        debug_assert!(
            cur_node.is_leaf() || cur_node.is_extension(),
            "Expecting leaf node or extension node. Current node tag: {}.",
            cur_node.tag()
        );

        // 1. Remove one or more nibbles from the old leaf/extension node and put it on one branch.
        //    If there's no matching nibble, we just need to remove one. Otherwise first remove all
        //    matching nibbles.
        let mut old_node_nibbler = Nibbler::new_with_is_odd(
            cur_node.encoded_path()?.clone(),
            cur_node.odd_encoded_path()?,
        );
        old_node_nibbler.advance(matching_nibbles.num_nibbles());
        let old_node_index: u8 = match old_node_nibbler.next() {
            Some(old_node_index) => old_node_index,
            _ => bail!("Nibbler ran out of nibbles"),
        };
        // Get the remaining nibbles and use them to construct a modified version of the old node.
        // Note that its encoded path may be empty, which means there was only one non-matching
        // nibble. In that case if it's an empty extension node we'll need to remove the node.
        let mut new_leaf_or_extension_node = cur_node.clone();
        {
            let nibbles = old_node_nibbler.get_all_remaining();
            new_leaf_or_extension_node.set_encoded_path(nibbles.encoded_path().to_vec())?;
            new_leaf_or_extension_node.set_odd_encoded_path(nibbles.is_odd())?;
        }
        let is_extension_node_empty = new_leaf_or_extension_node.is_extension()
            && new_leaf_or_extension_node
                .extension()?
                .encoded_path
                .is_empty();

        // 3. Create a new branch node with correct nibble height
        let mut new_branch_node = NodeType::Branch(BranchType::new());

        new_branch_node.set_child(
            old_node_index,
            if is_extension_node_empty {
                new_leaf_or_extension_node.extension()?.child
            } else {
                new_leaf_or_extension_node.hash()
            },
        )?;

        // 3. Insert a new leaf node on another branch.
        let new_leaf_index: u8 = match nibbler.next() {
            Some(new_leaf_index) => new_leaf_index,
            _ => bail!("Nibbler ran out of nibbles"),
        };
        debug_assert_ne!(
            new_leaf_index, old_node_index,
            "The new leaf node and the other node should not end up on the same branch.",
        );
        let inserted_child_hash =
            self.insert(&mut NodeType::default(), nibbler, insert_value, batch)?;
        new_branch_node.set_child(new_leaf_index, inserted_child_hash)?;

        // 4. If there were no matching nibbles, we are done.
        if matching_nibbles.is_empty() {
            batch.replace_node(&cur_node.hash(), new_branch_node.clone())?;
            if !is_extension_node_empty {
                batch.create_node(new_leaf_or_extension_node)?;
            }
            *cur_node = new_branch_node;
            return Ok(cur_node.hash());
        }

        // 5. If there were some matching nibbles, create a new extension node and connect it to
        //    the new branch node.
        let nibbles = matching_nibbles.get_all_remaining();
        let new_extension_node = NodeType::Extension(ExtensionType {
            encoded_path: nibbles.encoded_path().to_vec(),
            odd_encoded_path: nibbles.is_odd(),
            child: new_branch_node.hash(),
        });

        batch.replace_node(&cur_node.hash(), new_extension_node.clone())?;
        batch.create_node(new_branch_node)?;
        if !is_extension_node_empty {
            batch.create_node(new_leaf_or_extension_node)?;
        }
        *cur_node = new_extension_node;
        Ok(cur_node.hash())
    }

    /// Retrieves value by traversing the trie with given key with proof
    pub fn get_with_proof<T>(
        &self,
        mut nibbler: Nibbler,
        version: Version,
    ) -> Result<Option<(T, SparseMerkleProofDeprecated)>>
    where
        T: TryFrom<Vec<u8>>,
        <T as TryFrom<Vec<u8>>>::Error: ::failure::Fail,
    {
        ensure!(
            nibbler.num_all_remaining() as u8 == SHARD_ROOT_NIBBLE_HEIGHT,
            "Current implementation does not support variable nibbler remaining \
             length. nibbler key: {:?}, nibbler remaining length: {},  \
             shard_root_nibble_height: {}",
            nibbler.encoded_path(),
            nibbler.num_all_remaining(),
            SHARD_ROOT_NIBBLE_HEIGHT
        );

        // First, find root node at given version.
        let root_hash = get_root_hash_from_db(&self.db, version)?;

        let mut cur_node;
        let mut next_node = get_node_from_db(&self.db, &root_hash)?;

        let mut bitmap_nibbler =
            Nibbler::new_with_is_odd(vec![0x00; HashValue::LENGTH], true /* is_odd */);
        let mut non_default_siblings = vec![];
        let value: T;

        loop {
            cur_node = next_node;
            match cur_node {
                NodeType::Null(_) => {
                    return Ok(None);
                }
                NodeType::Branch(branch_node) => {
                    let index = match nibbler.next() {
                        Some(nibble) => nibble,
                        None => return Ok(None),
                    };
                    match branch_node.children.get(&index) {
                        Some(digest) => {
                            next_node = get_node_from_db(&self.db, digest)?;
                            let (bitmap_nibble, mut bitmap_nibble_non_default_siblings) =
                                branch_node.generate_bitmap_nibble_with_non_default_siblings(index);
                            bitmap_nibbler.set_next(bitmap_nibble);
                            bitmap_nibbler.next();
                            non_default_siblings.append(&mut bitmap_nibble_non_default_siblings);
                        }
                        None => return Ok(None),
                    };
                }
                NodeType::Extension(extension_node) => {
                    let mut extension_nibbler = Nibbler::new_with_is_odd(
                        extension_node.encoded_path.clone(),
                        extension_node.odd_encoded_path,
                    );
                    bitmap_nibbler.advance(
                        nibbler
                            .get_matching_nibbles(&mut extension_nibbler)
                            .num_nibbles(),
                    );
                    if !extension_nibbler.is_finished() {
                        return Ok(None);
                    }
                    next_node = get_node_from_db(&self.db, &extension_node.child)?;
                }
                NodeType::Leaf(leaf_node) => {
                    let mut leaf_nibbler = Nibbler::new_with_is_odd(
                        leaf_node.encoded_path.clone(),
                        leaf_node.odd_encoded_path,
                    );
                    nibbler.get_matching_nibbles(&mut leaf_nibbler);
                    if nibbler.is_finished() && leaf_nibbler.is_finished() {
                        // bitmap_nibbler could advance here. However,
                        // it is meaningless.  The proof definitely
                        // will not have any non-default digest in
                        // leaf node and the search reaches the end of
                        // the tree.
                        value = T::try_from(leaf_node.value.clone())?;
                        break;
                    } else {
                        return Ok(None);
                    }
                }
            }
        }
        // Some post-processing of proof:
        // Turn nibble_bitmap from vector to array
        let mut digest_bitmap = [0x00; HashValue::LENGTH];
        digest_bitmap.copy_from_slice(&bitmap_nibbler.encoded_path()[..HashValue::LENGTH]);
        Ok(Some((
            value,
            SparseMerkleProofDeprecated {
                // Ordered like nibbler
                bitmap: digest_bitmap,
                non_default_siblings,
            },
        )))
    }

    #[cfg(test)]
    pub fn get<T>(&self, nibbler: &Nibbler, version: Version) -> Result<Option<T>>
    where
        T: TryFrom<Vec<u8>>,
        <T as TryFrom<Vec<u8>>>::Error: ::failure::Fail,
    {
        match self.get_with_proof::<T>(nibbler.clone(), version)? {
            Some((value, _)) => Ok(Some(value)),
            None => Ok(None),
        }
    }

    /// This method will validate the data integrity of the subtree, whose
    /// root node is specified by root_digest.
    ///
    /// Returns Validity::Valid if the subtree is valid; Validity::Invalid if
    /// the subtree is invalid, so user could retrieve the whole subtree again;
    /// Validity::ConditionalValid(invalid_digests) if the subtree is valid on
    /// the condition that the nodes with invalid_digests are all valid but they
    /// failed to be validated locally. Under such scenario, user may want to
    /// just re-fetch smaller subtrees originating from them instead of the whole
    /// subtree to save network traffic.
    /// Note: if the number of invalid_digests exceeds a limit, thought rarely,
    /// due to memory allocation concern, we regard the whole subtree is invalid.
    pub fn validate(&self, root_digest: &HashValue) -> Validity {
        let mut invalid_digests = vec![];
        if self.internal_validate(root_digest, &mut invalid_digests) {
            if invalid_digests.is_empty() {
                Validity::Valid
            } else if invalid_digests.len() == 1 && *root_digest == invalid_digests[0] {
                // The root node is invalid
                Validity::Invalid
            } else {
                Validity::ConditionalValid(invalid_digests)
            }
        } else {
            Validity::Invalid
        }
    }

    /// Return false if the invalid_digests exceeds MAX_INVALID_NODE
    fn internal_validate(
        &self,
        root_digest: &HashValue,
        invalid_digests: &mut Vec<(HashValue)>,
    ) -> bool {
        let mut invalid = false;
        match get_node_from_db(&self.db, &root_digest) {
            Ok(root) => match root {
                NodeType::Branch(branch_node) => {
                    if *root_digest == branch_node.hash() {
                        for child in branch_node.children.values() {
                            if !self.internal_validate(child, invalid_digests) {
                                return false;
                            }
                        }
                    } else {
                        invalid = true;
                    }
                }
                NodeType::Extension(extension_node) => {
                    if *root_digest == extension_node.hash() {
                        if !self.internal_validate(&extension_node.child, invalid_digests) {
                            return false;
                        }
                    } else {
                        invalid = true;
                    }
                }
                NodeType::Leaf(leaf_node) => {
                    if *root_digest != leaf_node.hash() {
                        invalid = true;
                    }
                }
                NodeType::Null(_) => invalid = true,
            },
            Err(_) => invalid = true,
        }
        if invalid {
            invalid_digests.push(root_digest.clone());
        }
        invalid_digests.len() <= MAX_INVALID_NODE
    }
}

/// PatriciaMerkleSchema defines the requirements for any DB schema where the DB needs to store a
/// patricia merkle trie.
pub trait PatriciaMerkleSchema: Schema {
    fn to_patricia_merkle(self) -> Result<NodeType>;

    fn to_patricia_merkle_root(self) -> Result<HashValue>;

    fn to_latest_version(self) -> Result<Version>;

    fn to_patricia_merkle_schema(node_hash: HashValue, node: NodeType) -> Self;

    fn to_patricia_merkle_root_schema(version: Version, node_hash: HashValue) -> Self;

    fn to_latest_version_schema(version: Version) -> Self;
}
