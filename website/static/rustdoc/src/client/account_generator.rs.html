const MILLIS_PER_SEC: usize = 1_000;
const INITIAL_BALANCE: u64 = 1_000_000;

/// Account tracks a state of an account.
///
/// In our load generator, we create a number of Accounts, mint some balance for them and then try
/// to spend their balance one coin at a time. Since the validators will only process a transaction
/// with the correct sequence number, we need to take care to only have one outstanding transaction
/// from each account. Otherwise, the VM will fail the transaction.
struct Account {
    /// How many coins to mint.
    initial_balance: u64,
    /// Account data. This contains the sequence number for the account.
    source_account: AccountData,
    /// Source address.
    source_address: AccountAddress,
    /// Where to send the money.
    destination_address: AccountAddress,
    /// Latest balance we were able to gather from all the validators.
    current_balance: u64,
    /// How many transactions have been submitted. Difference between 'submitted' and
    /// 'current_balance' is how many transactions are in flight.
    submitted: u64,
    /// If the account has been minted.
    mint_state: MintState,
    /// Timestamp when we started minting.
    time_start_minting: Option<Instant>,
    /// Duration for minting.
    time_spent_minting: Duration,
    /// Timestamp for when we last sent a payment.
    time_start_last_transaction: Option<Instant>,
    /// Total cumulative time in payments.
    time_spent_transacting: Duration,
}

/// State to track if an account has been minted.
#[derive(PartialEq)]
enum MintState {
    /// Not yet minted. Need to send minting transaction.
    NotMinted,
    /// Minting transaction is sent to a validator, but balance is not yet reflected.
    WaitingForMint,
    /// Balance available to spend.
    Minted,
}

impl Account {
    fn new(balance: u64) -> Self {
        let source_account = Client::create_random_account().unwrap();
        let source_address = source_account.address;
        let destination_address = Client::create_random_account().unwrap().address;

        Account {
            initial_balance: balance,
            source_account,
            source_address,
            destination_address,
            current_balance: 0,
            submitted: 0,
            mint_state: MintState::NotMinted,
            time_start_minting: None,
            time_spent_minting: Duration::from_secs(0),
            time_start_last_transaction: None,
            time_spent_transacting: Duration::from_secs(0),
        }
    }

    /// Mint balance for the account.
    fn mint_balance(&mut self, sender: &mut AccountData, client: &Client) -> Option<impl Future> {
        if let MintState::NotMinted = self.mint_state {
            self.time_start_minting = Some(Instant::now());
            self.mint_state = MintState::WaitingForMint;
            Some(
                client
                    .mint_coins_async(sender, &mut self.source_account, self.initial_balance)
                    .expect("Error minting coins"),
            )
        } else {
            // Already minted or waiting for mint. Nothing more to do.
            None
        }
    }

    /// Spend one coin.
    ///
    /// Since the validators will fail transaction that doesn't have the correct sequence number, we
    /// can only have one outstanding transaction at a time. Spend a coin if we can.
    fn transfer_one(&mut self, client: &Client) -> Option<impl Future> {
        assert!(MintState::Minted == self.mint_state);
        if self.has_pending_transfer() {
            // Still waiting for a transaction to go through. Can't transfer.
            None
        } else {
            self.time_start_last_transaction = Some(Instant::now());
            self.submitted += 1;
            Some(
                client
                    .transfer_coins_async(&mut self.source_account, &self.destination_address, 1)
                    .expect("Error transfering coin"),
            )
        }
    }

    /// Update balance.
    ///
    /// We want to know the balance on all the validators in the network. The caller will provide a
    /// a vector of balances it sees for this account. We only update the account balance when all
    /// the validators have the same balance.
    fn update_balance(&mut self, mut validator_balances: Vec<u64>) -> u64 {
        let majority_count = std::cmp::min(
            validator_balances.len() * 2 / 3 + 1,
            validator_balances.len(),
        );

        validator_balances.sort_unstable();
        let mut curr_value = validator_balances[0];
        let mut curr_count = 0;

        for element in validator_balances.iter() {
            if *element == curr_value {
                curr_count += 1;
                if curr_count >= majority_count {
                    break;
                }
            } else {
                curr_count = 1;
                curr_value = *element;
            }
        }

        if curr_count >= majority_count && curr_value != 0 {
            // Only update balance when all validators agrees. In the future, we might be okay to
            // update when a greater than 2/3 majority agree.
            match self.mint_state {
                MintState::WaitingForMint => {
                    self.mint_state = MintState::Minted;
                    self.time_spent_minting = self
                        .time_start_minting
                        .expect("logic error: time_spent_minting is None")
                        .elapsed();
                }
                MintState::Minted => {
                    // Need to check min_balance != current_balance. There is a race condition where
                    // none of the validators have executed the transfer transactions yet so they'll
                    // agree on the balance.
                    if self.has_pending_transfer() && curr_value != self.current_balance {
                        self.time_spent_transacting += self
                            .time_start_last_transaction
                            .unwrap_or_else(|| {
                                panic!(
                                    "logic error: time_start_last_transaction is None. {} vs {} vs {}",
                                    self.initial_balance, self.submitted, self.current_balance
                                )
                            })
                            .elapsed();
                    }
                }
                MintState::NotMinted => (),
            }
            self.current_balance = curr_value;
        }
        self.current_balance
    }

    fn has_pending_transfer(&self) -> bool {
        self.initial_balance - self.submitted != self.current_balance
    }

    fn need_minting(&self) -> bool {
        self.mint_state == MintState::NotMinted
    }

    fn done_minting(&self) -> bool {
        self.mint_state == MintState::Minted
    }

    fn transactions_executed(&self) -> u64 {
        if self.done_minting() {
            self.initial_balance - self.current_balance
        } else {
            0
        }
    }
}

/// A loader is a thread of execution that drives the minting and transaction of a set of accounts.
struct Loader {
    /// ID for a loader.
    id: usize,
    /// The set of accounts the loader is driving.
    accounts: Vec<Account>,
}

impl Loader {
    fn new(id: usize, accounts: Vec<Account>) -> Self {
        Loader { id, accounts }
    }

    fn mint_accounts_from_genesis(
        genesis_account: &mut AccountData,
        num_accounts: usize,
        account_balance: u64,
        clients: &[Client],
    ) -> Vec<Account> {
        let mut accounts: Vec<Account> = vec![];

        // Mint accounts.
        //
        // We have some constraints when it comes to minting currently: on the validator side only
        // allow accounts with balance to mint other accounts. We also currently can only send one
        // transaction from an account at a time because sequence number needs to be sequential.
        //
        // We deal with problem by minting one account from geneis account, wait for the txn to
        // complete. Then we can use the newly minted account to mint more new accounts. For every
        // account we mint, they can be used as a sender to mint more accounts.
        while accounts.len() < num_accounts {
            let mut new_accounts = {
                let accounts_len = accounts.len();

                // Any minted account can be a sender.
                let mut senders = accounts
                    .iter_mut()
                    .map(|a| &mut a.source_account)
                    .collect::<Vec<&mut AccountData>>();
                if senders.is_empty() {
                    senders.push(genesis_account);
                }

                // Mint as many accounts as we have senders or until we have enough.
                let senders_len = senders.len();
                Self::mint_accounts(
                    senders,
                    account_balance,
                    std::cmp::min(senders_len, num_accounts - accounts_len),
                    clients,
                )
            };
            accounts.append(&mut new_accounts);
        }
        accounts
    }

    /// Mint `num_accounts` accounts. Minting requests is sent to `clients` in a round robin
    /// fashion.
    fn mint_accounts(
        mut available_senders: Vec<&mut AccountData>,
        account_balance: u64,
        num_accounts: usize,
        clients: &[Client],
    ) -> Vec<Account> {
        // We have three sets here.
        //
        // 1. a set of senders that are free to mint more accounts. A sender can only be used to
        // mint one account at a time.
        // 2. a set of accounts that need to be minted.
        // 3. a set of (sender, receiver) that's currently going through minting.
        //
        let mut accounts_to_mint = std::iter::repeat_with(|| Account::new(account_balance))
            .take(num_accounts)
            .collect::<Vec<_>>();
        let mut in_minting: Vec<(_, _)> = vec![];
        let mut minted_accounts = vec![];

        // used to round robin across all the validator.
        while minted_accounts.len() < num_accounts {
            // Update balance of all accounts in minting to see if they've finished.
            Self::update_balances_for_accounts(
                &mut in_minting.iter_mut().map(|v| &mut v.0).collect(),
                clients,
            );

            // Check which accounts are done minting. For accounts minted, remove them from
            // in_minting and put sender back into available_senders list.
            in_minting = in_minting
                .into_iter()
                .filter_map(|v| {
                    if v.0.done_minting() {
                        // sender is avaiable to send again.
                        available_senders.push(v.1);
                        minted_accounts.push(v.0);
                        // Return none to drop it from in_minting via filter_map.
                        None
                    } else {
                        Some(v)
                    }
                })
                .collect();

            // Gather accounts still to mint and available sender until either is exhauasted, then
            // mint them.
            let mut futures = vec![];
            while !accounts_to_mint.is_empty() && !available_senders.is_empty() {
                let mut account = accounts_to_mint
                    .pop()
                    .expect("logic error: accounts_to_mint can't be empty.");
                let sender = available_senders
                    .pop()
                    .expect("logic error: available_senders can't be empty.");

                if let Some(future) = account.mint_balance(sender, &clients[0]) {
                    futures.push(future);
                    in_minting.push((account, sender));
                }
            }

            println!(
                "{} Minting {} accounts, {} minted, {} in minting, {} to mint.",
                Local::now().to_rfc2822(),
                futures.len(),
                minted_accounts.len(),
                in_minting.len(),
                accounts_to_mint.len()
            );

            if !futures.is_empty() {
                // Drive all the futures to completion.
                if futures::future::join_all(futures).wait().is_err() {
                    panic!("Error minting coins!");
                }
            }
        }

        minted_accounts
    }

    /// Spend coins for up to `batch_size` accounts. Returns the number of transactions sent.
    fn transfer_coins(&mut self, clients: &[Client], batch_size: usize) -> usize {
        let mut i = 0;
        let futures = self
            .accounts
            .iter_mut()
            .filter_map(|a| {
                a.transfer_one(&clients[i % clients.len()]).map(|future| {
                    i += 1;
                    future
                })
            })
            .take(batch_size);
        if let Ok(results) = futures::future::join_all(futures).wait() {
            return results.len();
        }
        panic!("Failed to transfer coins!");
    }

    fn accounts_minted(&self) -> usize {
        self.accounts.iter().filter(|a| a.done_minting()).count()
    }

    fn accounts_minting_duration_total(&self) -> Duration {
        self.accounts.iter().map(|a| a.time_spent_minting).sum()
    }

    fn accounts_need_minting(&self) -> usize {
        self.accounts.iter().filter(|a| a.need_minting()).count()
    }

    fn accounts_transactions_executed(&self) -> u64 {
        self.accounts
            .iter()
            .map(Account::transactions_executed)
            .sum()
    }

    fn accounts_transactions_sent(&self) -> u64 {
        self.accounts.iter().map(|a| a.submitted).sum()
    }

    fn accounts_transactions_duration_total(&self) -> Duration {
        self.accounts.iter().map(|a| a.time_spent_transacting).sum()
    }

    fn update_balances(&mut self, clients: &[Client]) {
        // Only need to check accounts with outstanding transaction.
        let mut accounts_to_check = self
            .accounts
            .iter_mut()
            .filter(|a| !a.done_minting() || a.has_pending_transfer())
            .collect::<Vec<&mut Account>>();
        Self::update_balances_for_accounts(&mut accounts_to_check, clients)
    }

    /// Query all validators and update account balances.
    fn update_balances_for_accounts(accounts_to_check: &mut Vec<&mut Account>, clients: &[Client]) {
        let balances = {
            // We need to put the code block in {} to return 'accounts_to_check' when we are done.
            let addresses = accounts_to_check
                .iter()
                .map(|addr| addr.source_address)
                .collect::<Vec<_>>();
            let futures = clients.iter().map(|c| {
                c.get_balances_async(&addresses)
                    .expect("Failed to get balances from validator")
            });
            futures::future::join_all(futures)
                .wait()
                .expect("Failed to get balances.")
        };

        for b in balances.iter() {
            assert!(b.len() == accounts_to_check.len());
        }

        for (i, account) in accounts_to_check.iter_mut().enumerate() {
            let account_balances = balances.iter().map(|balances| balances[i]).collect();
            account.update_balance(account_balances);
        }
    }

    /// Tick is one iteration of the loader.
    ///
    /// In each tick, we:
    /// 1. update account balances according to 'operation'.
    /// 2. mint or transfer from accont up to 'batch_size'.
    /// 3. send some stats about this loader back through 'sender' .
    fn tick(
        &mut self,
        clients: &[Client],
        batch_size: usize,
        sender: &Sender<(usize, LoaderStats)>,
    ) {
        self.update_balances(clients);

        let sent = self.transfer_coins(clients, batch_size);
        if sent == 0 {
            // If we weren't able to send any transactions, it means all accounts for this
            // loader is busy.
            eprintln!("loader {}: all accounts have txns pending...", self.id);
        }

        let output = LoaderStats {
            total_txn_executed: self.accounts_transactions_executed(),
            total_txn_sent: self.accounts_transactions_sent(),
            total_minted: self.accounts_minted(),
            total_need_minting: self.accounts_need_minting(),
            total_minting_duration: self.accounts_minting_duration_total(),
            total_transaction_duraton: self.accounts_transactions_duration_total(),
        };

        if sender.send((self.id, output)).is_err() {
            return;
        }
    }

    /// Start a thread that runs the loader loop. Returns a channel receiver where LoaderStats are
    /// sent every iteration.
    pub fn start(
        mut self,
        clients: Vec<Client>,
        interval: Duration,
        target_txns_per_interval: usize,
    ) -> Receiver<(usize, LoaderStats)> {
        let (sender, receiver) = std::sync::mpsc::channel();
        std::thread::spawn(move || {
            let mut last_cycle_time = std::time::Instant::now() - interval;
            loop {
                let now = std::time::Instant::now();

                // We try to send target_txns_per_interval each interval. But sometimes it takes
                // longer to update all the balances and send all the transactions than the interval
                // time allows. In that case, we try to scale the bath size to make up for the
                // longer interval.
                let batch_size = target_txns_per_interval
                    * now.duration_since(last_cycle_time).as_millis() as usize
                    / interval.as_millis() as usize;

                self.tick(&clients, batch_size, &sender);

                last_cycle_time = now;

                let elapsed = now.elapsed();
                if elapsed < interval {
                    std::thread::sleep(interval - elapsed);
                } else {
                    eprintln!(
                        "Loader operation took longer than interval: {:?} vs {:?}",
                        elapsed, interval
                    );
                }
            }
        });
        receiver
    }
}

/// The current state of a single loader.
///
/// We want to be able to calculate sum and differences for these. So be sure to keep all the fields
/// commutative.
#[derive(Add, Sub, Default, Clone, Copy, Debug)]
pub struct LoaderStats {
    pub total_txn_executed: u64,
    pub total_txn_sent: u64,
    pub total_minted: usize,
    pub total_need_minting: usize,
    pub total_minting_duration: std::time::Duration,
    pub total_transaction_duraton: std::time::Duration,
}

/// Holds the LoaderStats from all the loaders and provide helper functions.
pub struct AggregatedLoaderStats {
    // We store a circular buffer that holds that last 60s worth of stats for each loader.
    stats: Vec<VecDeque<(Instant, LoaderStats)>>,
}

impl AggregatedLoaderStats {
    fn new(num_loaders: usize) -> Self {
        AggregatedLoaderStats {
            stats: std::iter::repeat_with(VecDeque::new)
                .take(num_loaders)
                .collect(),
        }
    }

    /// add a new data point.
    ///
    /// Assumes that timestamp is monotonic increasing for each index.
    fn add(&mut self, index: usize, timestamp: Instant, value: LoaderStats) {
        self.stats[index].push_back((timestamp, value));
    }

    /// expire any data points with time stamp older than cutoff
    fn expire_older_than(&mut self, cutoff: Instant) {
        for stats in self.stats.iter_mut() {
            while !stats.is_empty() && stats[0].0 < cutoff {
                stats.pop_front();
            }
        }
    }

    fn get_total(&self) -> LoaderStats {
        // Take the last value for each loader and sum them. This is the all time stats.
        self.stats
            .iter()
            .filter_map(|x| x.iter().last())
            .map(|x| x.1)
            .fold(Default::default(), |a, b| a + b)
    }

    fn get_total_at_instant(&self, instant: Instant) -> LoaderStats {
        self.stats
            .iter()
            .filter_map(|x| x.iter().find(|x| x.0 >= instant))
            .map(|x| x.1)
            .fold(Default::default(), |a, b| a + b)
    }

    fn get_delta_since(&self, instant: Instant) -> LoaderStats {
        self.get_total() - self.get_total_at_instant(instant)
    }
}

pub fn get_clients(addresses: &[&str]) -> Vec<(String, Client)> {
    let mut clients = Vec::new();
    for address in addresses {
        let host_port: Vec<&str> = address.split(',').collect();
        clients.push((
            address.to_string(),
            Client::new(&host_port[0], &host_port[1]),
        ));
    }
    clients
}

pub fn get_debug_clients(addresses: &[&str]) -> Vec<(String, NodeDebugInterfaceClient)> {
    let mut debug_clients = Vec::new();
    for address in addresses {
        let host_port: Vec<&str> = address.split(',').collect();
        let env = Arc::new(EnvBuilder::new().build());
        let ch = ChannelBuilder::new(env).connect(&format!("{}:{}", &host_port[0], &host_port[2],));
        debug_clients.push((address.to_string(), NodeDebugInterfaceClient::new(ch)));
    }
    debug_clients
}

pub fn run(
    addresses: Vec<&str>,
    num_loaders: usize,
    accounts_per_loader: usize,
    interval_millis: u64,
    target_tps: usize,
) {
    let clients: Vec<Client> = get_clients(&addresses).into_iter().map(|x| x.1).collect();

    let genesis_address = AccountAddress::from(&GENESIS_KEYPAIR.1);
    let mut sender = Client::create_account(
        genesis_address,
        GENESIS_KEYPAIR.0.clone(),
        GENESIS_KEYPAIR.1,
        0,
    );
    // Find the current sequence number for the genesis account.
    sender.sequence_number = clients[0].get_sequence_number(&genesis_address).unwrap();

    let all_accounts = Loader::mint_accounts_from_genesis(
        &mut sender,
        num_loaders * accounts_per_loader,
        INITIAL_BALANCE,
        &clients,
    );

    let mut accounts_iter = all_accounts.into_iter();

    let mut loaders = vec![];
    for i in 0..num_loaders {
        loaders.push(Loader::new(
            i,
            accounts_iter
                .by_ref()
                .take(accounts_per_loader)
                .collect::<Vec<_>>(),
        ));
    }

    let interval = Duration::from_millis(interval_millis);
    let per_loader_interval_target =
        target_tps / num_loaders * MILLIS_PER_SEC / interval_millis as usize;

    // Start the loaders and get channels to get their stats.
    let loader_receivers = loaders
        .into_iter()
        .map(|loader| {
            let clients = get_clients(&addresses).into_iter().map(|x| x.1).collect();
            loader.start(clients, interval, per_loader_interval_target)
        })
        .collect::<Vec<_>>();

    // gether the stats from the validator.
    let validator_clients = get_debug_clients(&addresses);
    let mut validator_views = validator_clients
        .into_iter()
        .map(|(conn_str, client)| (ValidatorStats::new(conn_str, client)))
        .collect::<Vec<_>>();

    let mut loader_stats = AggregatedLoaderStats::new(num_loaders);

    let start_transaction_time = Instant::now();

    loop {
        std::thread::sleep(std::time::Duration::from_millis(200));
        let now = Instant::now();

        // read all loader stats.
        for receiver in loader_receivers.iter() {
            if let Some((id, output)) = receiver.try_iter().last() {
                loader_stats.add(id, now, output);
            }
        }
        loader_stats.expire_older_than(now - Duration::from_secs(60));

        // read all validator stats.
        ValidatorStats::update_all(&mut validator_views);

        // Take the last value for each loader and sum them. This is the all time stats.
        let loader_total = loader_stats.get_total();

        let delta_10s = loader_stats.get_delta_since(now - Duration::from_secs(10));
        let delta_60s = loader_stats.get_delta_since(now - Duration::from_secs(60));

        let duration_transacting = start_transaction_time.elapsed();

        let content = ScreenContent {
            time_elapsed: start_transaction_time.elapsed(),
            duration_transacting,

            num_nodes: addresses.len(),
            num_loaders,
            accounts_per_loader,

            txns_last_10s: delta_10s.total_txn_executed,
            txns_last_60s: delta_60s.total_txn_executed,
            txns_all_time: loader_total.total_txn_executed,

            minted_last_10s: delta_10s.total_minted,
            minted_last_60s: delta_60s.total_minted,
            minted_all_time: loader_total.total_minted,

            aggregated_loader_stats: loader_total,
            validator_stats: validator_views
                .iter()
                .map(|stats| (stats.id.as_str(), stats.get_calculated_stats()))
                .collect(),
        };

        content.print_screen();
    }
}
