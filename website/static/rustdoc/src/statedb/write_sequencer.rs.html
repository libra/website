use crate::{
    shard::{StateDBShard, StateDBShardService},
    ShardId, StorageTransaction, NUM_SHARDS,
};
use crypto::HashValue;
use futures::lazy;
use promising_future::Promise;
use std::{
    collections::VecDeque,
    iter::repeat_with,
    sync::{
        atomic::{AtomicU64, Ordering},
        mpsc::{channel, Receiver, Sender},
        Arc, Mutex,
    },
};
use tokio_threadpool::{Builder, ThreadPool};
use types::transaction::Version;

/// Events the write sequencer accepts.
pub enum WriteSequencerEvent {
    /// A new commit to StateDB. This is sent by StateDb. We expect NUM_SHARDS number of
    /// `(StorageTransaction, Promise<HashValue>)`. The promise will be set after the transactions
    /// are committed to storage. The `HashValue` will be set to the new root hash of that shard.
    NewCommit(Version, Vec<(StorageTransaction, Promise<HashValue>)>),

    /// A write to a shard is done. This is sent by the worker.
    WriteDone(ShardId),

    /// A message that tells the sequencer to shut down.
    Shutdown,
}

/// WriteSequencer is responsible for sending writes to individual shards. The writes are done via a
/// thread pool. We only allow one write at a time per shard. Subsequent writes are put into a per
/// shard queue until the in-flight write is done.
///
/// We also maintain the current version of each shard, i.e., `statedb.versions`. Note that this
/// might be different from the version number in the DB, i.e., `StateSchema::LatestVersion`. In the
/// DB, we only update the shard's version number when a write touches the shard (the last version
/// the shard was written to). Here, we bump version number even if there is no write to a shard.
/// This is so that we know what the last version the shard can be read from safely.
///
/// For example, if a write at version 90 comes in that needs to write to shard 0 and 5, we will
/// send the write for shard 0 and 5 to the thread pool and update the version of any other shards
/// to 90. Now say a new write at version 91 wants to write to 0 and 6. We will update the version
/// for all shards but 0, 5, 6 to 91. The write to shard 6 is sent to thread pool. The write to
/// shard 0 is queued. An operation for updating version to 91 is queued for shard 5.
///
/// When write to shard 0 finishes, it'll bump its version to 90, and process the next write. When
/// the write to shard 5 finishes, it'll bump its version to 90, then see the next queued op for the
/// shard bump its version to 91.
pub struct WriteSequencer {
    /// Tracks if there is current ongoing write for a shard.
    inflight_write: [bool; NUM_SHARDS],

    /// Per shard queue for pending operations.
    pending_ops: Vec<VecDeque<(Version, StorageTransaction, Promise<HashValue>)>>,

    /// The current version each shard is up to date to for reads.
    versions: Arc<Vec<AtomicU64>>,

    /// Access to underlying DBs.
    shards: Arc<Vec<StateDBShardService>>,

    /// Channel for communicating with the sequencer.
    receiver: Receiver<WriteSequencerEvent>,

    /// The sender object here is only ever accessed from one thread. Sender
    /// needs to be in a Mutex because we share WriteSequencer object across
    /// threads.
    pub sender: Mutex<Sender<WriteSequencerEvent>>,

    /// Worker threads
    thread_pool: ThreadPool,
}

/// Operation indicate what operation we just performed in WriteSequencer::process_one().
enum OperationType {
    /// A write is dispatched to the storage.
    Write,
    /// Only version number is updated for the shard.
    VersionUpdate,
}

impl WriteSequencer {
    pub fn new(versions: Arc<Vec<AtomicU64>>, shards: Arc<Vec<StateDBShardService>>) -> Self {
        let (sender, receiver) = channel();

        WriteSequencer {
            inflight_write: [false; NUM_SHARDS],
            pending_ops: repeat_with(VecDeque::new).take(NUM_SHARDS).collect(),
            versions,
            shards,
            receiver,
            sender: Mutex::new(sender),
            thread_pool: Builder::new()
                .name_prefix("statedb")
                .pool_size(NUM_SHARDS)
                .build(),
        }
    }

    pub fn run(&mut self) {
        loop {
            let event = self.receiver.recv();

            match event.expect("sender hang up??") {
                // A new commit. If a shard is busy, add the transaction to its queue. Otherwise,
                // process it.
                WriteSequencerEvent::NewCommit(version, txns_and_promises) => {
                    debug_assert_eq!(txns_and_promises.len(), NUM_SHARDS);
                    for (shard, (txn, promise)) in txns_and_promises.into_iter().enumerate() {
                        if self.inflight_write[shard] {
                            self.pending_ops[shard].push_back((version, txn, promise));
                        } else if let OperationType::Write =
                            self.process_one(shard, version, txn, promise)
                        {
                            self.inflight_write[shard] = true;
                        }
                    }
                }
                // A write to a shard is done. Process queued operations until it's doing write
                // again or no more ops.
                WriteSequencerEvent::WriteDone(shard) => {
                    self.inflight_write[shard] = false;
                    while let Some((version, txn, promise)) = self.pending_ops[shard].pop_front() {
                        if let OperationType::Write = self.process_one(shard, version, txn, promise)
                        {
                            self.inflight_write[shard] = true;
                            break;
                        }
                    }
                }
                WriteSequencerEvent::Shutdown => break,
            }
        }
    }

    fn process_one(
        &mut self,
        shard: ShardId,
        version: Version,
        txn: StorageTransaction,
        promise: Promise<HashValue>,
    ) -> OperationType {
        let curr_version = self.versions[shard].load(Ordering::Acquire);
        assert!(version > curr_version);

        if txn.rows.is_empty() {
            // No write to this shard, just bump the version number and set the promise with the
            // previous root hash.
            self.versions[shard].store(version, Ordering::Release);

            // To keep the sequencer fast, off load reading of the root hash to the thread pool
            // since that needs to access rocksdb.
            let shards = Arc::clone(&self.shards);
            self.thread_pool.spawn(lazy(move || {
                // We want to return the root hash for the current version. Given the current
                // version didn't produce a write for this shard, we want the root hash for the last
                // version that had a write to the shard.
                //
                // At the time this closure is added to the thread pool, we know that version >
                // shards[shard].version(), but when the closure executes, shards[shard].version()
                // might have been updated. So we need to ask for the min of version,
                // shards[shard].version().
                let version_to_read = std::cmp::min(
                    version,
                    shards[shard].version().unwrap_or_else(|_| {
                        panic!("Unable to read version out of db shard {}", shard)
                    }),
                );

                let root_hash = shards[shard]
                    .get_digest(Some(version_to_read))
                    .unwrap_or_else(|_| panic!("Failed to get root hash from shard {}.", shard));
                promise.set(root_hash);
                Ok(())
            }));

            return OperationType::VersionUpdate;
        }

        let sender = self
            .sender
            .lock()
            .expect("Failed to obtain lock on sender.")
            .clone();
        let shards = Arc::clone(&self.shards);
        let versions = Arc::clone(&self.versions);

        self.thread_pool.spawn(lazy(move || {
            // TODO: handle write failure to DB.
            let root_hash = shards[shard]
                .put(txn)
                .unwrap_or_else(|_| panic!("Failed to write to shard {}.", shard));
            versions[shard].store(version, Ordering::Release);
            promise.set(root_hash);
            sender
                .send(WriteSequencerEvent::WriteDone(shard))
                .expect("Failed to send result back to WriteSequencer");
            Ok(())
        }));
        OperationType::Write
    }
}
